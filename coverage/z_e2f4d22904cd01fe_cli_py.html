1""" 

2dantinox CLI 

3============ 

4Entry point registered as the ``dantinox`` command by pyproject.toml. 

5 

6Subcommands 

7----------- 

8 train Train a model from a config file and a text corpus. 

9 generate Generate text from a trained checkpoint. 

10 sweep Run a W&B Bayesian hyperparameter sweep. 

11 benchmark Benchmark all (or selected) runs in a directory. 

12 

13Examples 

14-------- 

15 dantinox train --config configs/default_config.yaml --data_path data/corpus.txt 

16 dantinox generate --run_dir runs/run_20260101 --prompt "Nel mezzo del cammin " 

17 dantinox sweep --config configs/sweep.yaml --data_path data/corpus.txt 

18 dantinox benchmark --runs_dir runs --out_csv results.csv 

19"""

33def _add_config_overrides(parser: argparse.ArgumentParser) -> None: 

34 """Add one --<field> flag for every Config field.""" 

35 for field in dataclasses.fields(Config): 

36 flag = f"--{field.name}" 

37 if flag not in parser._option_string_actions: 

38 parser.add_argument(flag, type=type(field.default) if field.default is not dataclasses.MISSING else str, default=None) 

39

41def _apply_overrides(config: Config, args: argparse.Namespace) -> Config: 

42 """Write any non-None CLI overrides onto the config object.""" 

43 for field in dataclasses.fields(Config): 

44 val = getattr(args, field.name, None) 

45 if val is not None: 

46 setattr(config, field.name, val) 

47 return config

67def _cmd_generate(args: argparse.Namespace) -> None: 

68 import time 

69 

70 from dantinox.generator import Generator 

71 

72 gen = Generator(args.run_dir, seed=args.seed) 

73 

74 print(f"\nRun: {args.run_dir}") 

75 print(f"Prompt: {args.prompt}") 

76 print("-" * 40) 

77 

78 # Warmup / compile pass 

79 gen.generate(args.prompt, max_new_tokens=1) 

80 

81 t0 = time.time() 

82 text = gen.generate( 

83 args.prompt, 

84 max_new_tokens=args.max_new_tokens, 

85 greedy=args.greedy, 

86 top_k=args.top_k, 

87 top_p=args.top_p, 

88 temperature=args.temperature, 

89 use_cache=not args.no_cache, 

90 ) 

91 elapsed = time.time() - t0 

92 

93 prompt_tokens = len(gen.tokenizer.encode(args.prompt)) 

94 total_tokens = len(gen.tokenizer.encode(text)) 

95 new_tokens = total_tokens - prompt_tokens 

96 

97 print(text) 

98 print("-" * 40) 

99 print(f"Generated {new_tokens} tokens in {elapsed:.2f}s " 

100 f"({new_tokens / elapsed:.1f} tok/s)")

103def _cmd_sweep(args: argparse.Namespace) -> None: 

104 """Launch a W&B sweep agent using the existing train_sweep entry point.""" 

105 try: 

106 import wandb 

107 except ImportError: 

108 print("wandb is not installed. Install it with: pip install wandb", file=sys.stderr) 

109 sys.exit(1) 

110 

111 import yaml 

112 with open(args.sweep_config) as f: 

113 sweep_cfg = yaml.safe_load(f) 

114 

115 project = getattr(args, "wandb_project", None) or "DantinoX" 

116 sweep_id = wandb.sweep(sweep_cfg, project=project) # type: ignore[attr-defined] 

117 print(f"Sweep ID: {sweep_id} (project: {project})") 

118 

119 def _agent_fn() -> None: 

120 from dantinox.trainer import Trainer 

121 

122 run = wandb.init() # type: ignore[attr-defined] 

123 wc = dict(run.config) 

124 

125 base = Config.from_yaml(args.config) if args.config else Config() 

126 for k, v in wc.items(): 

127 if hasattr(base, k): 

128 setattr(base, k, v) 

129 

130 trainer = Trainer(base) 

131 trainer.fit(args.data_path, wandb_project=None) 

132 wandb.finish() # type: ignore[attr-defined] 

133 

134 wandb.agent(sweep_id, function=_agent_fn, count=getattr(args, "count", None)) # type: ignore[attr-defined]

137def _cmd_find_lr(args: argparse.Namespace) -> None: 

138 config = Config.from_yaml(args.config) 

139 config = _apply_overrides(config, args) 

140 

141 from dantinox.trainer import Trainer 

142 trainer = Trainer(config) 

143 suggested_lr, lr_hist, loss_hist = trainer.find_lr( 

144 args.data_path, 

145 min_lr=args.min_lr, 

146 max_lr=args.max_lr, 

147 num_steps=args.num_steps, 

148 ) 

149 print(f"\nSuggested learning rate: {suggested_lr:.2e}") 

150 if args.plot: 

151 try: 

152 import matplotlib.pyplot as plt 

153 fig, ax = plt.subplots(figsize=(8, 4)) 

154 ax.plot(lr_hist, loss_hist) 

155 ax.axvline(suggested_lr, color="red", linestyle="--", label=f"suggested: {suggested_lr:.2e}") 

156 ax.set_xscale("log") 

157 ax.set_xlabel("Learning rate") 

158 ax.set_ylabel("Smoothed loss") 

159 ax.set_title("LR Range Test") 

160 ax.legend() 

161 plt.tight_layout() 

162 out = args.plot_out or "lr_finder.png" 

163 fig.savefig(out, dpi=150) 

164 print(f"Plot saved to: {out}") 

165 except ImportError: 

166 print("matplotlib not installed — skipping plot (pip install matplotlib)")

217def _cmd_infbench(args: argparse.Namespace) -> None: 

218 """Delegate to benchmarks/run_all.py (subprocess keeps JAX state isolated).""" 

219 import subprocess 

220 from pathlib import Path 

221 

222 run_all = Path(__file__).resolve().parent.parent / "benchmarks" / "run_all.py" 

223 if not run_all.exists(): 

224 print(f"Error: {run_all} not found — is the repo intact?", file=sys.stderr) 

225 sys.exit(1) 

226 

227 cmd = [sys.executable, str(run_all), 

228 "--out-csv", args.out_csv, 

229 "--out-dir", args.out_dir, 

230 "--n-warmup", str(args.n_warmup), 

231 "--n-trials", str(args.n_trials)] 

232 if args.groups: 

233 cmd += ["--groups"] + args.groups 

234 if getattr(args, "device", None): 

235 cmd += ["--device", args.device] 

236 if args.sweep_only: 

237 cmd += ["--sweep-only"] 

238 if args.plot_only: 

239 cmd += ["--plot-only"] 

240 if args.verbose: 

241 cmd += ["--verbose"] 

242 if getattr(args, "trained", False): 

243 cmd += ["--trained"] 

244 if getattr(args, "inference_off", False): 

245 cmd += ["--inference-off"] 

246 if getattr(args, "runs_dir", None): 

247 cmd += ["--runs-dir", args.runs_dir] 

248 if getattr(args, "trained_csv", None): 

249 cmd += ["--trained-csv", args.trained_csv] 

250 if getattr(args, "trained_plot", None): 

251 cmd += ["--trained-plot", args.trained_plot] 

252 if getattr(args, "batch_csv", None): 

253 cmd += ["--batch-csv", args.batch_csv] 

254 if getattr(args, "batch_sizes", None): 

255 cmd += ["--batch-sizes"] + [str(b) for b in args.batch_sizes] 

256 if getattr(args, "batch_seq_len", None): 

257 cmd += ["--batch-seq-len", str(args.batch_seq_len)] 

258 

259 result = subprocess.run(cmd) 

260 sys.exit(result.returncode)
