51def _sample_logit( 

52 logits: jnp.ndarray, 

53 key: jax.Array, 

54 greedy: bool, 

55 temperature: float, 

56 top_k: int | None, 

57 top_p: float | None, 

58) -> tuple[int, jax.Array]: 

59 """Sample one token id from a [V] logit vector. Returns (token_id, new_key).""" 

60 log_probs = jax.nn.log_softmax(logits[0].astype(jnp.float32) / temperature) 

61 

62 if greedy: 

63 return int(jnp.argmax(log_probs)), key 

64 

65 if top_k is not None: 

66 top_k_vals, top_k_idx = jax.lax.top_k(jnp.exp(log_probs), top_k) 

67 filtered = jnp.full_like(log_probs, -jnp.inf) 

68 filtered = filtered.at[top_k_idx].set( 

69 jnp.log(top_k_vals / top_k_vals.sum() + 1e-10) 

70 ) 

71 log_probs = filtered 

72 

73 if top_p is not None: 

74 probs = jnp.exp(log_probs) 

75 sorted_idx = jnp.argsort(probs)[::-1] 

76 sorted_probs = probs[sorted_idx] 

77 cum = jnp.cumsum(sorted_probs) 

78 mask = (cum - sorted_probs) >= top_p 

79 filtered_p = jnp.where(mask, 0.0, sorted_probs) 

80 filtered_p = filtered_p / (filtered_p.sum() + 1e-10) 

81 filtered_lp = jnp.full_like(log_probs, -jnp.inf) 

82 filtered_lp = filtered_lp.at[sorted_idx].set(jnp.log(filtered_p + 1e-10)) 

83 log_probs = filtered_lp 

84 

85 new_key, subkey = jax.random.split(key) 

86 tok_id = int(jax.random.categorical(subkey, log_probs)) 

87 return tok_id, new_key

92def _load_checkpoint(run_dir: str, seed: int) -> tuple[Config, Transformer, Tokenizer]: 

93 """Return (config, model, tokenizer) loaded from a run directory.""" 

94 config_path = os.path.join(run_dir, "config.yaml") 

95 weights_path = os.path.join(run_dir, "model_weights.msgpack") 

96 

97 if not os.path.isdir(run_dir): 

98 raise CheckpointError(f"Run directory not found: {run_dir}") 

99 if not os.path.exists(config_path): 

100 raise CheckpointError(f"Config file not found: {config_path}") 

101 

102 with open(config_path) as f: 

103 raw = yaml.safe_load(f) 

104 

105 flat: dict = {} 

106 for section in raw.values(): 

107 if isinstance(section, dict): 

108 flat.update(section) 

109 if not flat: 

110 flat = raw 

111 

112 config = Config.from_dict(flat) 

113 

114 if config.mla: 

115 config.inference = True 

116 

117 tok_path = os.path.join(run_dir, "tokenizer.json") 

118 if os.path.exists(tok_path): 

119 tokenizer = load_tokenizer_from_file(tok_path) 

120 log.info("Loaded tokenizer from %s", tok_path) 

121 else: 

122 import warnings 

123 warnings.warn( 

124 f"tokenizer.json not found in {run_dir!r}. " 

125 "Falling back to re-training the tokenizer from the original corpus. " 

126 "Re-save your checkpoint with a newer version of DantinoX to avoid this.", 

127 DeprecationWarning, 

128 stacklevel=2, 

129 ) 

130 if config.dataset_source == "huggingface": 

131 from datasets import load_dataset 

132 raw_dataset = load_dataset(config.dataset_name, split="train") 

133 text = " ".join(raw_dataset["text"]) 

134 else: 

135 if not os.path.exists(config.dataset_name): 

136 raise CheckpointError( 

137 f"tokenizer.json not found and dataset file {config.dataset_name!r} " 

138 "is also missing. Cannot rebuild the tokenizer vocabulary." 

139 ) 

140 with open(config.dataset_name, encoding="utf-8") as f: 

141 text = f.read() 

142 lines = [line.rstrip() for line in text.split("\n") if line.strip()] 

143 blocks = ["\n".join(lines[i : i + 3]) for i in range(0, len(lines), 3)] 

144 text = "\n\n".join(blocks) + "\n" 

145 tokenizer = get_tokenizer(config.tokenizer_type) 

146 if config.tokenizer_type == "char": 

147 tokenizer.train_from_text(text) 

148 elif config.tokenizer_type == "bpe": 

149 tokenizer.train_from_text(text, vocab_size=config.vocab_size) 

150 

151 config.vocab_size = tokenizer.vocab_size 

152 

153 rngs = nnx.Rngs(seed) 

154 model = Transformer(config, rngs=rngs) 

155 

156 if os.path.exists(weights_path): 

157 log.info("Loading weights from %s", weights_path) 

158 with open(weights_path, "rb") as f: 

159 state_dict = msgpack.unpackb( 

160 f.read(), ext_hook=_msgpack_ext_unpack, strict_map_key=False 

161 ) 

162 nnx.update(model, state_dict) 

163 else: 

164 log.warning("No weights file found at %s — using random initialisation", weights_path) 

165 

166 return config, model, tokenizer

169# ── Generator ─────────────────────────────────────────────────────────────────

171class Generator: 

172 """ 

173 Loads a trained DantinoX checkpoint and generates text. 

174 

175 Parameters 

176 ---------- 

177 run_dir : str 

178 Path to a training run directory containing ``config.yaml`` and 

179 ``model_weights.msgpack``. 

180 seed : int 

181 RNG seed used for sampling (default 42). 

182 

183 Raises 

184 ------ 

185 CheckpointError 

186 If the run directory or config cannot be loaded. 

187 

188 Examples 

189 -------- 

190 >>> gen = Generator("runs/run_20260101_120000") 

191 >>> text = gen.generate("Nel mezzo del cammin ") 

192 >>> print(text) 

193 """ 

194 

195 def __init__(self, run_dir: str, *, seed: int = 42) -> None: 

196 self.run_dir = run_dir 

197 self.seed = seed 

198 self.config, self.model, self.tokenizer = _load_checkpoint(run_dir, seed) 

199 

200 def __repr__(self) -> str: 

201 attn = "MLA" if self.config.mla else ("GQA" if self.config.kv_heads < self.config.n_heads else "MHA") 

202 return f"Generator(run_dir={self.run_dir!r}, attn={attn}, seed={self.seed})" 

203 

204 def _bpe_fix(self, text: str) -> str: 

205 if self.config.tokenizer_type == "bpe": 

206 for old, new in _BPE_REPLACEMENTS: 

207 text = text.replace(old, new) 

208 return text

212 def generate( 

213 self, 

214 prompt: str, 

215 *, 

216 max_new_tokens: int = 150, 

217 greedy: bool = False, 

218 top_k: int | None = None, 

219 top_p: float | None = None, 

220 temperature: float = 1.0, 

221 use_cache: bool = True, 

222 ) -> str: 

223 """ 

224 Generate text continuing from ``prompt``. 

225 

226 Parameters 

227 ---------- 

228 prompt : str 

229 The input prefix. 

230 max_new_tokens : int 

231 Number of tokens to generate (default 150). 

232 greedy : bool 

233 Use greedy decoding instead of sampling (default False). 

234 top_k : int, optional 

235 Keep only the top-k logits before sampling. 

236 top_p : float, optional 

237 Nucleus sampling threshold. 

238 temperature : float 

239 Softmax temperature (default 1.0). 

240 use_cache : bool 

241 Enable KV-cache for faster generation (default True). 

242 

243 Returns 

244 ------- 

245 str 

246 The full generated string (prompt + continuation). 

247 """ 

248 tokens = self.tokenizer.encode(prompt) 

249 x = jnp.array([tokens], dtype=jnp.int32) 

250 

251 log.debug( 

252 "Generating %d tokens from prompt of %d tokens (greedy=%s, cache=%s)", 

253 max_new_tokens, len(tokens), greedy, use_cache, 

254 ) 

255 

256 output = _generate( 

257 model=self.model, 

258 x=x, 

259 max_generations=max_new_tokens, 

260 greedy=greedy, 

261 seed=self.seed, 

262 use_cache=use_cache, 

263 top_k=top_k, 

264 top_p=top_p, 

265 temperature=temperature, 

266 ) 

267 output.block_until_ready() 

268 

269 return self._bpe_fix(self.tokenizer.decode(output[0].tolist()))

271 # ── Batched generation ──────────────────────────────────────────────────── 

272 

273 def generate_batch( 

274 self, 

275 prompts: list[str], 

276 *, 

277 max_new_tokens: int = 150, 

278 greedy: bool = False, 

279 top_k: int | None = None, 

280 top_p: float | None = None, 

281 temperature: float = 1.0, 

282 use_cache: bool = True, 

283 ) -> list[str]: 

284 """ 

285 Generate text for multiple prompts in a single batched forward pass. 

286 

287 Shorter prompts are left-padded with zeros so all share the same 

288 sequence length. This runs a true batch through the model, so 

289 throughput scales with GPU parallelism. 

290 

291 Parameters 

292 ---------- 

293 prompts : list[str] 

294 Input prefixes to generate from. 

295 max_new_tokens : int 

296 Tokens to generate per prompt (default 150). 

297 greedy : bool 

298 Greedy decoding (default False). 

299 top_k : int, optional 

300 Top-k filtering before sampling. 

301 top_p : float, optional 

302 Nucleus sampling threshold. 

303 temperature : float 

304 Softmax temperature (default 1.0). 

305 use_cache : bool 

306 Enable KV-cache (default True). 

307 

308 Returns 

309 ------- 

310 list[str] 

311 Generated strings (prompt + continuation) in the same order as 

312 ``prompts``. 

313 """ 

314 if not prompts: 

315 return [] 

316 

317 encoded = [self.tokenizer.encode(p) for p in prompts] 

318 max_len = max(len(e) for e in encoded) 

319 

320 # Left-pad shorter prompts with zeros so all share the same start position. 

321 padded = [([0] * (max_len - len(e))) + e for e in encoded] 

322 x = jnp.array(padded, dtype=jnp.int32) # [B, max_len] 

323 

324 log.debug("Batch generating: B=%d max_prompt_len=%d max_new=%d", len(prompts), max_len, max_new_tokens) 

325 

326 output = _generate( 

327 model=self.model, 

328 x=x, 

329 max_generations=max_new_tokens, 

330 greedy=greedy, 

331 seed=self.seed, 

332 use_cache=use_cache, 

333 top_k=top_k, 

334 top_p=top_p, 

335 temperature=temperature, 

336 ) 

337 output.block_until_ready() 

338 

339 results = [] 

340 for i, enc in enumerate(encoded): 

341 # Strip the left-padding: prompt starts at (max_len - len(enc)) 

342 start = max_len - len(enc) 

343 tokens_out = output[i, start:].tolist() 

344 results.append(self._bpe_fix(self.tokenizer.decode(tokens_out))) 

345 return results

347 # ── Streaming generation ────────────────────────────────────────────────── 

348 

349 def stream( 

350 self, 

351 prompt: str, 

352 *, 

353 max_new_tokens: int = 150, 

354 greedy: bool = False, 

355 top_k: int | None = None, 

356 top_p: float | None = None, 

357 temperature: float = 1.0, 

358 ) -> Iterator[str]: 

359 """ 

360 Stream generated tokens one at a time as they are produced. 

361 

362 Uses the KV-cache path: the prompt is prefilled in one forward pass, 

363 then each new token is decoded individually. Each ``yield`` returns 

364 the string for one generated token (may be a character or a BPE 

365 subword). 

366 

367 Parameters 

368 ---------- 

369 prompt : str 

370 The input prefix. 

371 max_new_tokens : int 

372 Maximum number of tokens to generate (default 150). 

373 greedy : bool 

374 Greedy decoding (default False). 

375 top_k : int, optional 

376 Top-k filtering. 

377 top_p : float, optional 

378 Nucleus sampling threshold. 

379 temperature : float 

380 Softmax temperature (default 1.0). 

381 

382 Yields 

383 ------ 

384 str 

385 Decoded string for each generated token. 

386 

387 Examples 

388 -------- 

389 >>> gen = Generator("runs/my_run") 

390 >>> for chunk in gen.stream("Nel mezzo", max_new_tokens=50): 

391 ... print(chunk, end="", flush=True) 

392 """ 

393 tokens = self.tokenizer.encode(prompt) 

394 T = len(tokens) 

395 max_ctx = self.config.max_context # type: ignore[attr-defined] 

396 num_blocks = self.config.num_blocks # type: ignore[attr-defined] 

397 

398 # Build full-context input with prompt at the start. 

399 x = jnp.zeros((1, max_ctx), dtype=jnp.int32) 

400 x = x.at[0, :T].set(jnp.array(tokens, dtype=jnp.int32)) 

401 

402 init_kv_cache = tuple((None, None) for _ in range(num_blocks)) 

403 key = jax.random.key(self.seed) 

404 

405 # Prefill: one pass over the entire prompt, populate KV cache. 

406 logits, kv_cache = _stream_prefill(self.model, x, init_kv_cache) 

407 

408 # Sample the first generated token from the last prompt position. 

409 tok_id, key = _sample_logit(logits[:, T - 1, :], key, greedy, temperature, top_k, top_p) 

410 yield self._bpe_fix(self.tokenizer.decode([tok_id])) 

411 

412 # Autoregressive decode loop. 

413 for pos in range(T, T + max_new_tokens - 1): 

414 if pos >= max_ctx: 

415 break 

416 tok = jnp.array([[tok_id]], dtype=jnp.int32) 

417 logits, kv_cache = _stream_decode(self.model, tok, kv_cache, jnp.array(pos)) 

418 tok_id, key = _sample_logit(logits[:, 0, :], key, greedy, temperature, top_k, top_p) 

419 yield self._bpe_fix(self.tokenizer.decode([tok_id]))
